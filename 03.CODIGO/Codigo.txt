function doGet() {
  return HtmlService.createTemplateFromFile('Index')
      .evaluate()
      .setTitle('Sistema de Seguridad - Control Interno')
      .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL);
}

function include(filename) {
  return HtmlService.createHtmlOutputFromFile(filename).getContent();
}

function getSheets() {
  return SpreadsheetApp.getActiveSpreadsheet();
}

// Registro de Usuario
function registrarUsuario(username, email, area) {
  var sheet = getSheets().getSheetByName('Usuarios');
  var data = sheet.getDataRange().getValues();
  
  for(var i=1; i<data.length; i++) {
    if(data[i][0] === username || data[i][1] === email) {
      return {success: false, message: 'El usuario o email ya está registrado.'};
    }
  }
  
  var tempPass = 'Seg' + Math.random().toString(36).substring(2, 7).toUpperCase() + '!';
  sheet.appendRow([username, email, tempPass, area, true]);
  
  try {
    MailApp.sendEmail({
      to: email,
      subject: 'Acceso al Sistema de Seguridad - Contraseña Temporal',
      htmlBody: '<h3>Bienvenido</h3><p>Tu usuario ha sido creado para el área: <b>'+area+'</b></p><p>Contraseña temporal: <b>'+tempPass+'</b></p>'
    });
    return {success: true, message: 'Usuario registrado con éxito. Revisa el correo.'};
  } catch(e) {
    return {success: false, message: 'Registrado, pero ocurrió un error al enviar el correo.'};
  }
}

// Login
function loginUsuario(username, password) {
  var sheet = getSheets().getSheetByName('Usuarios');
  var data = sheet.getDataRange().getValues();
  
  for(var i=1; i<data.length; i++) {
    if(data[i][0] === username && data[i][2] === password) {
      return {
        success: true,
        user: { username: data[i][0], email: data[i][1], area: data[i][3], esTemporal: data[i][4] }
      };
    }
  }
  return {success: false, message: 'Credenciales incorrectas.'};
}

function actualizarPassword(username, nuevaPassword) {
  var sheet = getSheets().getSheetByName('Usuarios');
  var data = sheet.getDataRange().getValues();
  for(var i=1; i<data.length; i++) {
    if(data[i][0] === username) {
      sheet.getRange(i+1, 3).setValue(nuevaPassword);
      sheet.getRange(i+1, 5).setValue(false);
      return {success: true};
    }
  }
  return {success: false, message: 'Error de actualización.'};
}

function guardarInforme(fecha, hora, area, codigo, texto, usuario) {
  var sheet = getSheets().getSheetByName('Informes');
  var id = 'INF-' + new Date().getTime();
  sheet.appendRow([id, fecha, hora, area, codigo, texto, usuario]);
  return {success: true, message: 'Informe guardado correctamente.'};
}

function getNotificacionesMonitoreo(usuario) {
  var sheetInf = getSheets().getSheetByName('Informes');
  var sheetResp = getSheets().getSheetByName('Respuestas');
  var informes = sheetInf.getDataRange().getValues();
  var respuestas = sheetResp.getDataRange().getValues();
  var listado = [];
  
  for(var i=1; i<informes.length; i++) {
    if(informes[i][6] === usuario || usuario === 'Administrador') { 
      var infId = informes[i][0].toString();
      for(var j=1; j<respuestas.length; j++) {
        if(respuestas[j][0].toString() === infId) {
          listado.push({
            id: infId,
            codigo: informes[i][4],
            area: respuestas[j][1],
            respuesta: respuestas[j][2]
          });
        }
      }
    }
  }
  return listado;
}

function getNotificacionesArea(areaViendo) {
  var sheetInf = getSheets().getSheetByName('Informes');
  var sheetResp = getSheets().getSheetByName('Respuestas');
  var informes = sheetInf.getDataRange().getValues();
  var respuestas = sheetResp.getDataRange().getValues();
  
  var respondidos = respuestas.map(function(r) { return r[0].toString(); });
  var listado = [];
  
  for(var i=1; i<informes.length; i++) {
    var idInforme = informes[i][0].toString();
    var areaDestino = informes[i][3];
    
    if (respondidos.indexOf(idInforme) !== -1) continue;
    
    if (areaDestino === areaViendo) {
      var fechaRaw = informes[i][1];
      var fechaStr = fechaRaw instanceof Date ? fechaRaw.toISOString().substring(0, 10) : fechaRaw.toString();
      listado.push({
        id: idInforme,
        fecha: fechaStr,
        hora: informes[i][2].toString(),
        codigo: informes[i][4].toString(),
        texto: informes[i][5].toString()
      });
    }
  }
  return listado;
}

function responderNotificacion(idInforme, area, texto, usuario) {
  var sheet = getSheets().getSheetByName('Respuestas');
  sheet.appendRow([idInforme, area, texto, usuario]);
  return {success: true, message: 'Respuesta enviada.'};
}

// BÚSQUEDA CORREGIDA INDEPENDIENTE
function buscarInformes(fechaDesde, fechaHasta, horaDesde, horaHasta, codigo) {
  var sheet = getSheets().getSheetByName('Informes');
  var data = sheet.getDataRange().getValues();
  var resultados = [];
  
  var tieneFiltros = (fechaDesde || fechaHasta || horaDesde || horaHasta || codigo);
  if (!tieneFiltros) return [];

  for(var i=1; i<data.length; i++) {
    var fechaRaw = data[i][1];
    var f = fechaRaw instanceof Date ? fechaRaw.toISOString().substring(0, 10) : fechaRaw.toString();
    var h = data[i][2].toString(); 
    var cod = data[i][4].toString();
    
    var cumple = true;
    
    // Filtro de Fecha (Rango o exacta)
    if(fechaDesde && f < fechaDesde) cumple = false;
    if(fechaHasta && f > fechaHasta) cumple = false;
    
    // Filtro de Hora independiente (Formato militar 00:00)
    if(horaDesde && h < horaDesde) cumple = false;
    if(horaHasta && h > horaHasta) cumple = false;
    
    // Filtro de Código estricto
    if(codigo && !cod.includes(codigo)) cumple = false;
    
    if(cumple) {
      resultados.push({
        id: data[i][0].toString(),
        fecha: f,
        hora: h,
        area: data[i][3],
        codigo: cod,
        texto: data[i][5].toString()
      });
    }
  }
  return resultados;
}
